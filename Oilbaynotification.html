        <script>

          const COUNTRY_CODE = "965"; // hardcoded for Kuwait

          const MESSAGE = "Your oil change is completed. Please proceed to the bay.";

          // generate 10 rows

          for (let i = 0; i < 10; i++) {

            document.write(`

              <tr>

                <td><input type="text" placeholder="Plate No"></td>

                <td><input type="tel" placeholder="Mobile Number"></td>

                <td>

                  <button class="whatsapp" onclick="openWhatsApp(this)">

                    WHATSAPP

                  </button>

                </td>

                <td>

                  <button class="clear" onclick="clearRow(this)">

                    CLEAR

                  </button>

                </td>

              </tr>

            `);

          }

          // open WhatsApp chat with ready message

          function openWhatsApp(btn) {

            const row = btn.parentElement.parentElement;

            let phone = row.children[1].children[0].value.trim();

            if (!phone) {

              alert("Please enter phone number");

              return;

            }

            // keep only digits

            phone = phone.replace(/[^0-9]/g, "");

            if (phone.length != 8) { // 8 digits for Kuwait numbers

              alert("Enter 8-digit mobile number only (no country code)");

              return;

            }

            const fullNumber = COUNTRY_CODE + phone;

            // encode message for URL

            const encodedMessage = encodeURIComponent(MESSAGE);

            // open WhatsApp link in new tab

            const waLink = `https://wa.me/${fullNumber}?text=${encodedMessage}`;

            window.open(waLink, "_blank");

          }

          // clear row inputs

          function clearRow(btn) {

            const row = btn.parentElement.parentElement;

            row.querySelectorAll("input").forEach(input => input.value = "");

          }
